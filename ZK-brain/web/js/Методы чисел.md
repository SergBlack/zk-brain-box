# Методы чисел
```js
num.toString(base); // возвращает строковое представление числа num в системе счисления base

num.toFixed(precision); // округляет число до precisionзнаков после запятой (аналогично Math.round) и возвращает строку. 
// toFixed(n) можно использовать для обхода проблемы с неточностью
const result = 0.1 + 0.2; // 0.30000000000000004
result.toFixed(2) // 0.30

isNaN(num); // преобразует аргумент в число и проверяет является ли оно NaN
isFinite(num); // преобразует аргумент в число и возвращает true, если оно является обычным числом, т.е. не NaN/Infinity/-Infinity

// получить числовое значение из строк, в которых первые знаки числа
parseInt(string, radix); // вернет `NaN`, если не смог прочитать ни одну цифру, radix определяет систему счисления
parseFloat(string); // вернет `NaN`, если не смог прочитать ни одну цифру

// методы Math для работы с числами
Math.floor(num); // округление в меньшую сторону
Math.ceil(num); // округление в большую сторону
Math.round(num); // округление в ближайшего целого
Math.trunc(num); // удаление дробной части без округления
Math.random(); // возвращает псевдослучайное число в диапазоне от 0 (включительно) до 1 (но не включая 1)
Math.max(a, b, c...)/Math.min(a, b, c...) // возвращает наибольшее/наименьшее число из перечисленных аргументов
Math.pow(n, power) // возвращает число n, возведённое в степень power
```
